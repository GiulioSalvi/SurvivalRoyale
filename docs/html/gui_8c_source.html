246PageData* getPageData(int maxRows, int maxColumns, Player** players, int totalPlayers, int* totalPages, int* bestStartColumn) {
247 // Constants for card layout
248 int columnSpacing = CARD_WIDTH + 5; // Card width + spacing between cards
249 int rowSpacing = CARD_HEIGHT + 3; // Card height + name row + life points row + minimum space row
250
251 // Calculate the height of the upper section
252 int upperSectionHeight = maxRows - (1 + 1 + LOG_SECTION_HEIGHT + 1 + 1); // // Upper line + upper minimal space + log section + line above
253
254 // Determine usable rows in the upper section
255 int availableRows = upperSectionHeight / rowSpacing;
256
257 // Adjust usable columns for horizontal centering
258 int usableColumns = maxColumns - 2; // Subtract minimal padding
259 int displayablePlayersPerRow = usableColumns / columnSpacing;
260
261 // Calculate the remaining space for centering cards horizontally
262 int totalCardWidth = displayablePlayersPerRow * columnSpacing;
263 int remainingSpace = usableColumns - totalCardWidth;
264 *bestStartColumn = 3 + (remainingSpace / 2); // Center cards horizontally
265
266 // Total players per page
267 int playersInEachFilledPage = availableRows * displayablePlayersPerRow;
268 *totalPages = (totalPlayers + playersInEachFilledPage - 1) / playersInEachFilledPage;
269
270 // Allocate memory for pages
271 PageData* pagesData = (PageData*)malloc(sizeof(PageData) * (*totalPages));
272 if (!pagesData) {
273 exit(EXIT_ALLOC_FAILURE);
274 }
275
276 int playersArrayIndex = 0; // Track players assigned to pages
277
278 for (int i = 0; i < *totalPages; i++) {
279 int playersOnThisPage = (playersArrayIndex + playersInEachFilledPage > totalPlayers)
280 ? totalPlayers - playersArrayIndex
281 : playersInEachFilledPage;
282
283 // Fill page data
284 pagesData[i].playerRows = availableRows;
285 pagesData[i].playerPerRow = displayablePlayersPerRow;
286 pagesData[i].playerCount = playersOnThisPage;
287 pagesData[i].rowSpacing = rowSpacing;
288
289 pagesData[i].players = (Player**)malloc(sizeof(Player*) * playersOnThisPage);
290 if (!pagesData[i].players) {
291 exit(EXIT_ALLOC_FAILURE);
292 }
293
294 for (int j = 0; j < playersOnThisPage; j++) {
295 pagesData[i].players[j] = players[playersArrayIndex++];
296 }
297 }
298
299 return pagesData;
300}
